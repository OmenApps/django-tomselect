class MultiTypeFeaturedAdapterView(AutocompleteIterablesView):
    """Subviews are dispatched via as_view() (mirrors _delegate_value).

    First-page only: each subview is called once and its results are merged.
    has_more=False keeps the contract with the frontend (which needs both
    has_more AND next_page truthy to paginate).
    """
    skip_authorization = True

    def get(self, request, *args, **kwargs):
        scope = request.GET.get("scope")
        rows = []
        for key, type_label, view_cls, label_field in routes():
            if scope and scope != key:
                continue
            try:
                resp = view_cls.as_view()(request)
            except PermissionDenied:
                continue
            for r in json.loads(resp.content).get("results", []):
                pk = r.get((view_cls.value_fields or ["id"])[0], r.get("id"))
                rows.append(sanitize_dict({
                    "value": f"{key}:{pk}",
                    "label": str(r.get(label_field, "")),
                    "_type_key": key,
                    "_type_label": type_label,
                }))
        return JsonResponse({"results": rows, "page": 1, "has_more": False, "next_page": None})
